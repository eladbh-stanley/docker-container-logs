🎯 Quick Start Commands

Docker Containers

# Monitor specific container
npm start -- --container web-app
# Monitor all containers
npm start -- --all

Log Files

# Analyze existing file
npm start -- --file app.log
# Live tailing
npm start -- --file app.log --follow

Remote Logs

# Remote URL
npm start -- --url https://logs.com/app.log
# Compressed logs
npm start -- --url logs.gz
